Для успешной регистрации с подтвержением по Email
надо поместить Properties.json файл по следующему пути: 
PCAccounting\WebClient\Properties\Properties.json

{
  "SendersName": "Имя отправителя",
  "MyEmail": "Email отправителя",
  "PasswordForMyEmail": "пароль",
  "AddressSMTP": "smtp аддресс",
  "PortSMTP": smtp порт
}
